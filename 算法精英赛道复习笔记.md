## 一、复杂度分析

| 符号 | 含义 | 示例 |
|------|------|------|
| O(1) | 常数时间 | 数组随机访问 |
| O(log n) | 对数时间 | 二分查找 |
| O(n) | 线性时间 | 遍历数组 |
| O(n log n) | 线性对数 | 快速排序、归并排序 |
| O(n²) | 平方时间 | 冒泡排序、双层循环 |
| O(2ⁿ) | 指数时间 | 暴力枚举子集 |

**估算技巧**：1 秒约运行 10^8 次操作，n 的范围决定可用算法：

| n 范围 | 可接受复杂度 |
|--------|------------|
| n ≤ 20 | O(2ⁿ), O(n!) |
| n ≤ 10³ | O(n²) |
| n ≤ 10⁵ | O(n log n) |
| n ≤ 10⁷ | O(n) |
| n > 10⁷ | O(log n) 或 O(1) |

## 二、基础算法

### 排序

**快速排序（qsort 手写版）**
```c
void qsort(int a[], int l, int r) {
    if (l >= r) return;
    int i = l, j = r, p = a[l + (r - l) / 2];
    while (i <= j) {
        while (a[i] < p) i++;
        while (a[j] > p) j--;
        if (i <= j) { int t = a[i]; a[i] = a[j]; a[j] = t; i++; j--; }
    }
    qsort(a, l, j);
    qsort(a, i, r);
}
```

**归并排序（求逆序对）**
```c
long long ans = 0;
void merge(int a[], int l, int m, int r) {
    int i = l, j = m + 1, k = 0;
    int *t = (int*)malloc((r - l + 1) * sizeof(int));
    while (i <= m && j <= r) {
        if (a[i] <= a[j]) t[k++] = a[i++];
        else { t[k++] = a[j++]; ans += m - i + 1; }
    }
    while (i <= m) t[k++] = a[i++];
    while (j <= r) t[k++] = a[j++];
    for (i = l, k = 0; i <= r; i++, k++) a[i] = t[k];
    free(t);
}
void mergeSort(int a[], int l, int r) {
    if (l >= r) return;
    int m = l + (r - l) / 2;
    mergeSort(a, l, m);
    mergeSort(a, m + 1, r);
    merge(a, l, m, r);
}
```

### 搜索

**二分查找（标准模板）**
```c
// 查找第一个 >= target 的位置
int lower_bound(int a[], int n, int target) {
    int l = 0, r = n;
    while (l < r) {
        int m = l + (r - l) / 2;
        if (a[m] >= target) r = m;
        else l = m + 1;
    }
    return l;
}

// 查找第一个 > target 的位置
int upper_bound(int a[], int n, int target) {
    int l = 0, r = n;
    while (l < r) {
        int m = l + (r - l) / 2;
        if (a[m] > target) r = m;
        else l = m + 1;
    }
    return l;
}

// 实数二分（保留 k 位小数）
double binary_search_real(double l, double r, double target) {
    for (int i = 0; i < 100; i++) {  // 迭代足够多次
        double m = (l + r) / 2;
        if (f(m) < target) l = m;
        else r = m;
    }
    return l;
}
```

**Dijkstra（堆优化）**
```c
// 需用优先队列，竞赛中可用数组模拟或直接 C++。C 语言下常用 SPFA 或朴素 Dijkstra
int dist[100010], vis[100010];
void dijkstra(int s, int n) {
    memset(dist, 0x3f, sizeof(dist));
    memset(vis, 0, sizeof(vis));
    dist[s] = 0;
    for (int i = 0; i < n; i++) {
        int u = -1, min = 0x3f3f3f3f;
        for (int j = 1; j <= n; j++)
            if (!vis[j] && dist[j] < min) { min = dist[j]; u = j; }
        if (u == -1) break;
        vis[u] = 1;
        for (int j = head[u]; j != -1; j = nxt[j]) {
            int v = to[j];
            if (!vis[v] && dist[v] > dist[u] + w[j])
                dist[v] = dist[u] + w[j];
        }
    }
}
```

**Floyd（多源最短路）**
```c
for (int k = 1; k <= n; k++)
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++)
            if (dist[i][j] > dist[i][k] + dist[k][j])
                dist[i][j] = dist[i][k] + dist[k][j];
```

### 数论

**最大公约数（辗转相除法）**
```c
int gcd(int a, int b) { return b == 0 ? a : gcd(b, a % b); }
int lcm(int a, int b) { return a / gcd(a, b) * b; }
```

**扩展欧几里得（求 ax + by = gcd(a, b) 的一组解）**
```c
int exgcd(int a, int b, int *x, int *y) {
    if (b == 0) { *x = 1; *y = 0; return a; }
    int g = exgcd(b, a % b, y, x);
    *y -= a / b * *x;
    return g;
}
```

**快速幂**
```c
long long qpow(long long a, long long b, long long m) {
    long long r = 1;
    while (b) {
        if (b & 1) r = r * a % m;
        a = a * a % m;
        b >>= 1;
    }
    return r;
}
```

**素数筛（埃氏筛）**
```c
int is_prime[1000010];
void sieve(int n) {
    for (int i = 2; i <= n; i++) is_prime[i] = 1;
    for (int i = 2; i * i <= n; i++)
        if (is_prime[i])
            for (int j = i * i; j <= n; j += i)
                is_prime[j] = 0;
}
```

**线性筛（更快）**
```c
int primes[1000010], cnt = 0, vis[1000010];
void linear_sieve(int n) {
    for (int i = 2; i <= n; i++) {
        if (!vis[i]) primes[cnt++] = i;
        for (int j = 0; j < cnt && i * primes[j] <= n; j++) {
            vis[i * primes[j]] = 1;
            if (i % primes[j] == 0) break;
        }
    }
}
```

```c
    if (k < 0 || k > n) return 0;
    return fac[n] * inv[k] % mod * inv[n - k] % mod;
```

### 概率 & 统计

- **期望** = Σ(值 × 概率)
- **前缀和**：`sum[i] = sum[i-1] + a[i]`，区间和 `[l, r] = sum[r] - sum[l-1]`
- **差分**：`d[l] += x; d[r+1] -= x;` 后求前缀和得到原数组加 x 后的结果
- **二维前缀和**：`S[i][j] = S[i-1][j] + S[i][j-1] - S[i-1][j-1] + a[i][j]`

## 五、高级算法

### 动态规划

**0-1 背包**
```c
for (int i = 0; i < n; i++)
    for (int c = capacity; c >= w[i]; c--)
        dp[c] = max(dp[c], dp[c - w[i]] + v[i]);
// 初始化：dp[0] = 0，其余 -inf（恰好装满）或 0（不超过）
```

**完全背包**
```c
for (int i = 0; i < n; i++)
    for (int c = w[i]; c <= capacity; c++)
        dp[c] = max(dp[c], dp[c - w[i]] + v[i]);
```

**最长递增子序列 LIS（O(n log n)）**
```c
int len = 0;
for (int i = 0; i < n; i++) {
    int p = lower_bound(dp, len, a[i]);  // 第一个 >= a[i] 的位置
    dp[p] = a[i];
    if (p == len) len++;
}
// len 即为 LIS 长度
```

**最长公共子序列 LCS**
```c
for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++)
        if (a[i-1] == b[j-1]) dp[i][j] = dp[i-1][j-1] + 1;
        else dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
```

**区间 DP 模板**
```c
for (int len = 2; len <= n; len++)          // 区间长度
    for (int i = 1; i + len - 1 <= n; i++) { // 左端点
        int j = i + len - 1;                  // 右端点
        for (int k = i; k < j; k++)
            dp[i][j] = max(dp[i][j], dp[i][k] + dp[k+1][j] + cost);
    }
```

**状压 DP 模板**
```c
// n ≤ 20 时使用
for (int mask = 0; mask < (1 << n); mask++)
    for (int i = 0; i < n; i++)
        if (!(mask & (1 << i)))
            dp[mask | (1 << i)] = max(dp[mask | (1 << i)], dp[mask] + ...);
```

**树形 DP**
```c
void dfs(int u, int fa) {
    // dp[u][0] = 不选 u, dp[u][1] = 选 u
    for (int i = head[u]; i != -1; i = nxt[i]) {
        int v = to[i];
        if (v == fa) continue;
        dfs(v, u);
        dp[u][0] += max(dp[v][0], dp[v][1]);
        dp[u][1] += dp[v][0];
    }
}
```

### 贪心

**典型场景**：
- **区间调度**：按结束时间排序，选不冲突的
- **哈夫曼编码**：每次取最小两个合并（用优先队列）
- **活动选择**：最早结束的优先
- **最小生成树**：Kruskal（按边权排序）+ 并查集
- **分饼干**：最小饼干满足最小胃口

### 分治

**经典应用**：
- 归并排序
- 快速排序
- 最近点对
- 大整数乘法（Karatsuba）
- 最大子段和（跨越中间 + 左右递归）

```c
int maxSubArray(int a[], int l, int r) {
    if (l == r) return a[l];
    int m = l + (r - l) / 2;
    int left = maxSubArray(a, l, m);
    int right = maxSubArray(a, m + 1, r);
    int cross = a[m], sum = a[m];
    for (int i = m - 1; i >= l; i--) { sum += a[i]; if (sum > cross) cross = sum; }
    sum = cross;
    for (int i = m + 1; i <= r; i++) { sum += a[i]; if (sum > cross) cross = sum; }
    if (left >= right && left >= cross) return left;
    if (right >= cross) return right;
    return cross;
}
```

### 位运算

| 操作 | 代码 | 含义 |

| 第 k 位 | `x >> k & 1` | 取 x 的第 k 位 |
